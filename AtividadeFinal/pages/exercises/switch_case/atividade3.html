            if (document.getElementById('numint').value == "") {

            } else {
                let num1 = parseInt(document.getElementById('numint').value);

                    case 0:
                    case 1:
                    case 2:
                    case 3:
                        console.log("Bebê");
                        return false;
                        break;

                    case 5:

                    case 8:
                    case 9:
                    case 10:
                        console.log("Criança");
                        return false;
                        break;

                    case 12:
                    case 13:
                    case 14:
                    case 15:
                    case 16:
                    case 17:
                    case 18:
                        console.log("Adolescente");
                        return false;
                        break;

                    default:
                        console.log("Adulto");
                        return false;
                        break;
                }
            }
        }
